package main

import (
	"fmt"
	"os"

	"github.com/MYORG/plugin-{{ .ServiceName }}/data"
	"github.com/MYORG/plugin-{{ .ServiceName }}/evaluation_plans"

	"github.com/privateerproj/privateer-sdk/command"
	"github.com/privateerproj/privateer-sdk/pluginkit"
)

var (
	// Version is to be replaced at build time by the associated tag
	Version = "0.0.0"
	// VersionPostfix is a marker for the version such as "dev", "beta", "rc", etc.
	VersionPostfix = "dev"
	// GitCommitHash is the commit at build time
	GitCommitHash = ""
	// BuiltAt is the actual build datetime
	BuiltAt = ""

	PluginName   = "{{ .ServiceName }}"
	RequiredVars = []string{}
)

func main() {
	if VersionPostfix != "" {
		Version = fmt.Sprintf("%s-%s", Version, VersionPostfix)
	}

	orchestrator := pluginkit.EvaluationOrchestrator{
		PluginName:    PluginName,
		PluginVersion: Version,
		PluginUri:     "",
	}

	catalog, err := data.Get{{ simplifiedName .Catalog.Metadata.Id .Catalog.Metadata.Version }}Catalog()
	if err != nil {
		fmt.Printf("Error loading OSPS Baseline catalog: %v\n", err)
		os.Exit(1)
	}

	catalog.Metadata.Id = "{{ .Catalog.Metadata.Id }}"

	orchestrator.AddEvaluationSuite(data.Loader, evaluation_plans.{{ .Catalog.Metadata.Id | snake_case }}, &catalog)

	runCmd := command.NewPluginCommands(
		PluginName,
		Version,
		VersionPostfix,
		GitCommitHash,
		&orchestrator,
	)

	err = runCmd.Execute()
	if err != nil {
		os.Exit(1)
	}
}
